{{/* Resolve only images explicitly named in ordinary image/images front matter. */}} {{- $page := . -}} {{- $candidates := slice -}} {{- with $page.Params.image -}} {{- if reflect.IsSlice . -}} {{- range . -}} {{- $candidates = $candidates | append . -}} {{- end -}} {{- else -}} {{- $candidates = $candidates | append . -}} {{- end -}} {{- end -}} {{- with $page.Params.images -}} {{- if reflect.IsSlice . -}} {{- range . -}} {{- $candidates = $candidates | append . -}} {{- end -}} {{- else -}} {{- $candidates = $candidates | append . -}} {{- end -}} {{- end -}} {{- $images := slice -}} {{- range $candidates -}} {{- $candidate := strings.TrimSpace (printf "%v" .) -}} {{- $url := "" -}} {{- if $candidate -}} {{- if strings.HasPrefix $candidate "https://" -}} {{- $url = $candidate -}} {{- else if strings.HasPrefix $candidate "http://" -}} {{- warnf "Ignoring insecure image %q on %s; use an HTTPS URL" $candidate $page.Path -}} {{- else -}} {{- with $page.Resources.GetMatch $candidate -}} {{- $url = .Permalink -}} {{- else -}} {{- if strings.HasPrefix $candidate "/" -}} {{- $url = urls.AbsURL $candidate -}} {{- else -}} {{- with resources.Get $candidate -}} {{- $url = .Permalink -}} {{- else -}} {{- warnf "Ignoring unresolved image %q on %s; use a page resource, global asset, root-relative static path, or absolute URL" $candidate $page.Path -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- if and $url (not (in $images $url)) -}} {{- $images = $images | append $url -}} {{- end -}} {{- end -}} {{- return $images -}}
